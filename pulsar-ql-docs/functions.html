Functions Reference

DOM Functions

HTML/XML document manipulation and web scraping functions

Document Loading and Parsing

DOM.load(url)

Load page from database cache or fetch from web if not cached/expired.

SELECT DOM.load('https://example.com');
DOM.fetch(url)

Fetch page immediately from web, bypassing cache.

SELECT DOM.fetch('https://example.com');
DOM.parse(html)

Parse HTML content string into DOM document.

SELECT DOM.parse('Hello');

Element Selection

DOM.select(dom, cssQuery)

Select all elements matching CSS selector.

SELECT DOM.select(dom, 'div.content');
DOM.selectFirst(dom, cssQuery)

Select first element matching CSS selector.

SELECT DOM.selectFirst(dom, 'title');
DOM.selectNth(dom, cssQuery, n)

Select nth element matching CSS selector.

SELECT DOM.selectNth(dom, 'p', 2);

Content Extraction

DOM.text(dom)

Extract text content from element.

SELECT DOM.text(DOM.selectFirst(dom, 'h1'));
DOM.attr(dom, attrName)

Get element attribute value.

SELECT DOM.attr(DOM.selectFirst(dom, 'a'), 'href');
DOM.hrefs(dom)

Extract all link hrefs from document.

SELECT DOM.hrefs(dom) as all_links;

String Functions

Text processing, validation, and manipulation functions

String Validation

STR.isNumeric(text)

Check if string contains only numeric characters (0-9).

SELECT STR.isNumeric('12345'); -- returns true
STR.isAlpha(text)

Check if string contains only alphabetic characters.

SELECT STR.isAlpha('hello'); -- returns true
STR.isEmpty(text)

Check if string is null or empty.

SELECT STR.isEmpty(''); -- returns true

String Manipulation

STR.capitalize(text)

Capitalize first character of string.

SELECT STR.capitalize('hello'); -- returns 'Hello'
STR.substring(text, start, end)

Extract substring from string.

SELECT STR.substring('hello world', 0, 5); -- returns 'hello'
STR.replace(text, search, replacement)

Replace occurrences of substring.

SELECT STR.replace('hello world', 'world', 'universe');

Array Functions

Array manipulation and utility operations

ARRAY.joinToString(array, separator)

Join array elements into string with separator.

SELECT ARRAY.joinToString(makeArray('a', 'b', 'c'), ','); -- returns 'a,b,c'
ARRAY.firstNotBlank(array)

Find first non-blank string value in array.

SELECT ARRAY.firstNotBlank(makeArray('', '  ', 'hello')); -- returns 'hello'
ARRAY.firstNotEmpty(array)

Find first non-empty string value in array.

SELECT ARRAY.firstNotEmpty(makeArray('', 'hello', 'world')); -- returns 'hello'

DateTime Functions

Date and time extraction and formatting

TIME.firstDateTime(text, pattern)

Extract first datetime from text using pattern.

SELECT TIME.firstDateTime('Meeting: 2023-12-25 14:30:00');
TIME.firstMysqlDateTime(text)

Extract MySQL-formatted datetime from text.

SELECT TIME.firstMysqlDateTime('Created at 2023-12-25 14:30:00');

Metadata Functions

Web page metadata and information retrieval

META.get(url)

Load and format web page metadata as string.

SELECT META.get('https://example.com');

Common Functions

Utility functions for URLs, regex, and data conversion

getTopPrivateDomain(url)

Extract top private domain from URL.

SELECT getTopPrivateDomain('https://www.example.com/path'); -- returns 'example.com'
re1(text, regex, group)

Extract regex capturing group from text.

SELECT re1('Price: $99.99', '\$([0-9.]+)'); -- returns '99.99'
formatTimestamp(timestamp, format)

Format timestamp to readable date string.

SELECT formatTimestamp('1640995200000', 'yyyy-MM-dd'); -- returns '2022-01-01'
